(function(factory) {
    if (typeof define === 'function' && define.amd) {
        // AMD
        define(['jquery'], factory);
    } else if (typeof exports === 'object') {
        // CommonJS
        factory(require('jquery'));
    } else {
        // Browser globals
        factory(jQuery);
    }
}(function(jQuery) {
    jQuery.fn.gituiNav = function(options) {
        var opt = {
        defaultExpand: 1, //默认展开第几项 0为不展开
        defaultExpand: 1, //默认展开第几项 0为不展开
        defaultExpand: 1, //默认展开第几项 0为不展开
        defaultExpand: 1, //默认展开第几项 0为不展开
            defaultExpand: 1, //默认展开第几项 0为不展开
            defaultActive: 1 //默认第几个选中
        };
        var opts = jQuery.extend({}, opt, options);
        //绑定click事件
        var navMain = jQuery(".gitui-nav");
        var navItem = jQuery(".gitui-nav-item");
        navItem.children("a").on('click', function(event) {
            if (!jQuery(this).parent().hasClass("active")) {
                //添加点击效果
                jQuery(this).parent().addClass("active");
                //移除其他效果
                jQuery(this).parent().siblings().removeClass("active");
                //隐藏其他菜单
                jQuery(this).parent().siblings().find("ul").slideUp("fast");
                //显示该菜单
                jQuery(this).next().slideDown("fast");
            } else {
                jQuery(this).parent().removeClass("active");
                jQuery(this).next().slideUp("fast");
            }
        });
        //是否有菜单需要展开
    };
}));
